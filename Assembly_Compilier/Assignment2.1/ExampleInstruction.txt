# sets the memory address 0 to store the value 1. M stands for memory.
M:0=1,1=1,2=5
# All programs starts with an initial setup of values in memory such as the example shown above
0 LOAD 0,0 #stores 1 in register 0
1 LOAD 1,1 #stores 1 in register 1
2 LOAD 2,2 #stores 5 in register 2
3 ADD 0,1
4 LSS 0,2,3 #checks to see if contents of register 0 (the sum) is less than register 2 (5), if true, jump to 3
5 STOR 0,3 #stores register 0 in memory address 3
6 STOP
